% ======================================================
%  Cheat Sheet Layout Definition (safe modern version)
% ======================================================

% --- Required packages ---
\usepackage{xcolor}
\usepackage{tikz}
\usepackage[framemethod=tikz]{mdframed}
\usepackage{tcolorbox}
\usepackage{fancyhdr}
\usepackage{titlesec}
\usepackage{microtype}

% --- Colors and styles ---
\definecolor{myblue}{cmyk}{1,.72,0,.38}
\colorlet{circle edge}{myblue}
\colorlet{circle area}{myblue!5}

\tcbset{
  cheatbox/.style={
    enhanced,
    colback=white,
    fontupper=\sffamily,
    colframe=gray,
    boxrule=0.5pt,
    arc=2pt,
    left=4pt,right=4pt,top=2pt,bottom=2pt,
  }
}

% --- Venn diagram circle styles (if used) ---
\tikzset{
  filled/.style={fill=circle area, draw=circle edge, thick},
  outline/.style={draw=circle edge, thick}
}

% --- Make math blue everywhere ---
\everymath\expandafter{\the\everymath \color{myblue}}
\everydisplay\expandafter{\the\everydisplay \color{myblue}}

% --- Compact layout ---
\renewcommand{\baselinestretch}{.8}
\setlength{\parindent}{0pt}

% --- Header / Footer setup ---
\pagestyle{fancy}
\fancyhf{}
\fancyhead[C]{\footnotesize\sffamily Hilfszettel zur Klausur\quad--\quadSeite \thepage}
\renewcommand{\headrulewidth}{0.4pt}
\fancyfoot{}

% --- Section formatting ---
\titleformat*{\section}{\color{myblue}\sffamily\small\bfseries}
\titleformat*{\subsection}{\sffamily\bfseries}
\titleformat*{\subsubsection}{\sffamily\itshape}

% --- Optional small spacing before/after sections ---
\titlespacing*{\section}{0pt}{.3ex}{.3ex}
\titlespacing*{\subsection}{0pt}{.2ex}{.2ex}

% --- Optional helper command for headers inside columns ---
\newcommand{\headerbox}[1]{%
  \begin{mdframed}[linecolor=gray,linewidth=0.5pt,
                   leftmargin=0mm,rightmargin=0mm,
                   skipbelow=0mm,skipabove=1mm]
  \footnotesize\sffamily #1
  \end{mdframed}
}
